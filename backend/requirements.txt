# ---- TTS engines ----
# Both engines are optional at runtime — if one fails to import, only the
# other is offered in the engine picker. Install both by default.

# F5-TTS (Apache 2.0, EN + ZH).
f5-tts>=0.3

# Coqui TTS, for XTTS-v2 (CPML non-commercial, 17 languages).
# Set COQUI_TOS_AGREED=1 in your env to accept their license on model download
# (tts.py does this automatically). The package is heavy (~500 MB of deps incl.
# torchaudio, librosa, numba).
TTS>=0.22

# Pin transformers to a version that still exposes BeamSearchScorer.
# Coqui TTS (now unmaintained) imports BeamSearchScorer from transformers,
# which was **removed entirely in transformers 5.0**. Without this pin pip
# picks up a 5.x release (TTS 0.22.0 only pins `transformers>=4.33.0` with
# no upper bound) and the XTTS engine fails at import with
# `cannot import name 'BeamSearchScorer'`.
# 4.46.3 is the highest 4.x line that tests clean against TTS 0.22.0; 4.36.2
# also works but misses unrelated bug fixes. Re-test before bumping.
transformers==4.46.3

# PyTorch 2.6 flipped `torch.load(weights_only=True)` as the default; Coqui
# TTS 0.22.0 still calls `torch.load()` without the flag, so the XTTS
# checkpoint refuses to unpickle. `backend/tts.py::_XTTSEngine._load` installs
# a narrow monkey-patch that forces `weights_only=False` for the XTTS load
# path (safe because checkpoints come from the Coqui HF mirror, not user
# uploads). No upper pin needed on torch — the patch is forward-compatible.

# ---- Deep Clone (Stage 1 dataset prep) ----
# Whisper is used to transcribe 3-10 s training segments, producing the
# metadata.csv expected by F5-TTS's finetune_cli.py and XTTS's trainer.
# faster-whisper ships CTranslate2 wheels that work cleanly on Apple Silicon
# (no CUDA required); it's ~4x faster than openai-whisper for equal accuracy.
faster-whisper>=1.0.3
